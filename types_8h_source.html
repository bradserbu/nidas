53 /*
54  * A data sample as it is passed from kernel-level drivers
55  * to user space.
56  *
57  * The data member array length is 0, allowing one to create
58  * varying length samples.
59  * In actual use one will create and use a dsm_sample
60  * as follows:
61  struct dsm_sample* samp =
62  kmalloc(SIZEOF_DSM_SAMPLE_HEADER + SPACE_ENOUGH_FOR_DATA,GFP_KERNEL);
63  ...
64  samp->timetag = xxx;
65  if (len > SPACE_ENOUGH_FOR_DATA) we_ve_got_trouble();
66  samp->length = len;
67  memcpy(samp->data,buffer,len);
68  ...
69  *
70  * When sample is read on the user side:
71  struct dsm_sample header;
72  read(fd,&header,SIZEOF_DSM_SAMPLE_HEADER);
73  char* data = (char*) malloc(header.length);
74  * read data portion
75  read(fd,data,header.length);
76  */
