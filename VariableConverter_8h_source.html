278 inline double Polynomial::eval(double x,float *p, unsigned int np)
279 {
280  double y = 0.0;
281  if (np == 0) return y;
282  for (unsigned int i = np - 1; i > 0; i--) {
283  y += p[i];
284  y *= x;
285  }
286  y += p[0];
287  return y;
288 }
