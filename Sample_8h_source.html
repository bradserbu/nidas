479  // Technically speaking, I think it is unsafe to call the subclass
480  // method getType() when calling the base class constructor, because
481  // the subclass has not been initialized yet. It works here probably
482  // because GCC binds directly to the subclass getSampleType() method,
483  // probably because of some obscure C++ scoping rules. If it didn't,
484  // then the call would fail since the VTable has not been fully
485  // initialized yet, or if it is it is initialized with the base class
486  // bindings, for which getType() is pure virtual.
487 
488  // Therefore the getType() call has been replaced with getSampleType().
489  // That seems safer as long as that call only depends upon on the
490  // operand type and not on its value, since its value is an
491  // uninitialized pointer member.
492  //
493  // An alternative to calling a function at all is to use type traits
494  // templates, such as defined in sample_traits.h:
495  //
496  // Sample(sample_type_traits<DataT>::sample_type_enum)

713 template <class DataT>
714 void SampleT<DataT>::freeReference() const
715 {
716  // if refCount is 0, put it back in the Pool.
717 #ifdef USE_ATOMIC_REF_COUNT
718  // GCC 4.X atomic operations
719  int rc = __sync_sub_and_fetch(&_refCount,1);
720  assert(rc >= 0);
721  if (rc == 0)
722  SamplePool<SampleT<DataT> >::getInstance()->putSample(this);
723 #else
724 #ifdef MUTEX_PROTECT_REF_COUNTS
725  _refLock.lock();
726 #endif
727  bool ref0 = --_refCount == 0;
728  assert(_refCount >= 0);
729 #ifdef MUTEX_PROTECT_REF_COUNTS
730  _refLock.unlock();
731 #endif
732  if (ref0)
733  SamplePool<SampleT<DataT> >::getInstance()->putSample(this);
734 #endif
735 }
