890 template<class SocketT>
891 int McSocketMulticaster<SocketT>::run() throw(Exception)
892 {
893  MulticastSocket* requestmsock = 0;
894 
895  int sockfd = (_serverSocket ? _serverSocket->getFd() :
896  _datagramSocket->getFd());
897 
898 #ifdef HAVE_PPOLL
899  struct pollfd fds;
900  fds.fd = sockfd;
901 #ifdef POLLRDHUP
902  fds.events = POLLIN | POLLRDHUP;
903 #else
904  fds.events = POLLIN;
905 #endif
906 #else
907  fd_set fdset;
908  FD_ZERO(&fdset);
909 #endif
910 
911 
912  // wait for this amount of time after a datagram send to receive a
913  // connection or receipt of UDP data
914  struct timespec timeout;
915  timeout.tv_sec = 0;
916  timeout.tv_nsec = NSECS_PER_SEC / 4; // portion of a second
917 
918  // get the existing signal mask
919  sigset_t sigmask;
920  pthread_sigmask(SIG_BLOCK,NULL,&sigmask);
921  // unblock SIGUSR1 in pselect/ppoll
922  sigdelset(&sigmask,SIGUSR1);
923 
924  Inet4SocketAddress mcsockaddr =
925  _mcsocket->getInet4McastSocketAddress();
926  Inet4Address mcaddr = mcsockaddr.getInet4Address();
927 
928  std::vector<Inet4NetworkInterface> ifaces;
929 
930  if (mcaddr.isMultiCastAddress()) {
931  _requestSocket = requestmsock = new MulticastSocket();
932  if (_mcsocket->getInterface().getAddress() == Inet4Address(INADDR_ANY)) {
933  std::list<Inet4NetworkInterface> tmpifaces = requestmsock->getInterfaces();
934  std::list<Inet4NetworkInterface>::const_iterator ifacei = tmpifaces.begin();
935  for ( ; ifacei != tmpifaces.end(); ++ifacei) {
936  Inet4NetworkInterface iface = *ifacei;
937  int flags = iface.getFlags();
938  if (flags & IFF_UP && flags & IFF_BROADCAST && flags & (IFF_MULTICAST | IFF_LOOPBACK))
939  ifaces.push_back(iface);
940  }
941  }
942  }
943  else
944  _requestSocket = new DatagramSocket();
945 
946  McSocketDatagram dgram;
947  dgram.setMagic(dgram.magicVal);
948  dgram.setSocketAddress(mcsockaddr);
949  dgram.setRequestType(_mcsocket->getRequestType());
950  dgram.setSocketType(getMcSocketType(_mcsocket));
951  if (_serverSocket)
952  dgram.setRequesterListenPort(_serverSocket->getLocalPort());
953  else
954  dgram.setRequesterListenPort(_datagramSocket->getLocalPort());
955 
956  for (int numCasts=0; !isInterrupted() ; numCasts++) {
957  // If multicast, loop over interfaces
958  try {
959  if (requestmsock && ifaces.size() > 0) {
960  Inet4NetworkInterface iface = ifaces[numCasts % ifaces.size()];
961  requestmsock->setInterface(mcaddr,iface);
962  _requestSocket->send(dgram);
963  }
964  else _requestSocket->send(dgram);
965  if (!(numCasts % 10))
966  std::cerr << "sent " << numCasts << " dgrams" <<
967  ", requestType=" << dgram.getRequestType() <<
968  ", port=" << dgram.getRequesterListenPort() <<
969  ", socketType=" << dgram.getSocketType() <<
970  ", len=" << dgram.getLength() <<
971  ", #mcifaces=" << ifaces.size() << std::endl;
972  }
973  catch(const IOException& e) {
974  // perhaps the interface has disappeared. Log the error.
975  WLOG(("McSocketMulticaster: %s: %s",
976  _requestSocket->getLocalSocketAddress().toString().c_str(),e.what()));
977  sleep(10);
978  if (requestmsock) {
979  _requestSocket->close();
980  delete _requestSocket;
981  // close and re-create the socket
982  _requestSocket = requestmsock = new MulticastSocket();
983  // re-create the list of interfaces.
984  ifaces.clear();
985  std::list<Inet4NetworkInterface> tmpifaces = requestmsock->getInterfaces();
986  std::list<Inet4NetworkInterface>::const_iterator ifacei = tmpifaces.begin();
987  for ( ; ifacei != tmpifaces.end(); ++ifacei) {
988  Inet4NetworkInterface iface = *ifacei;
989  int flags = iface.getFlags();
990  if (flags & IFF_UP && flags & IFF_BROADCAST && flags & (IFF_MULTICAST | IFF_LOOPBACK))
991  ifaces.push_back(iface);
992  }
993  }
994  continue;
995  }
996 
997  int res;
998 #ifdef HAVE_PPOLL
999  res = ::ppoll(&fds,1,&timeout,&sigmask);
1000 
1001  if (res > 0) {
1002 
1003  if (fds.revents & POLLERR) {
1004  _mcsocketMutex.lock();
1005  if (_mcsocket) _mcsocket->offer(errno);
1006  _mcsocketMutex.unlock();
1007  if (_serverSocket) _serverSocket->close(); // OK to close twice
1008  if (_datagramSocket) _datagramSocket->close();
1009  _requestSocket->close();
1010  throw IOException("McSocket","ppoll",errno);
1011  }
1012 
1013 #ifdef POLLRDHUP
1014  if (fds.revents & (POLLHUP | POLLRDHUP))
1015 #else
1016  if (fds.revents & POLLHUP)
1017 #endif
1018  WLOG(("%s POLLHUP","McSocket"));
1019 
1020  }
1021 
1022 #else
1023  assert(sockfd >= 0 && sockfd < FD_SETSIZE); // FD_SETSIZE=1024
1024  FD_SET(sockfd, &fdset);
1025  res = ::pselect(sockfd+1,&fdset,0,0,&timeout,&sigmask);
1026 #endif
1027  if (res <= 0) {
1028  if (res == 0) continue;
1029  if (errno == EINTR) break;
1030  _mcsocketMutex.lock();
1031  if (_mcsocket) _mcsocket->offer(errno);
1032  _mcsocketMutex.unlock();
1033  if (_serverSocket) _serverSocket->close(); // OK to close twice
1034  if (_datagramSocket) _datagramSocket->close();
1035  _requestSocket->close();
1036  throw IOException("McSocket","poll/select",errno);
1037  }
1038 
1039  if (_serverSocket) {
1040  Socket* socket = _serverSocket->accept();
1041  DLOG(("accepted socket connection from ") <<
1042  socket->getRemoteSocketAddress().toString() << " on " <<
1043  _serverSocket->getLocalSocketAddress().toString());
1044  _serverSocket->close();
1045  nidas::util::Inet4PacketInfoX pktinfo;
1046  if (socket->getRemoteSocketAddress().getFamily() == AF_INET) {
1047  nidas::util::Inet4SocketAddress remoteAddr =
1048  nidas::util::Inet4SocketAddress((const struct sockaddr_in*)
1049  socket->getRemoteSocketAddress().getConstSockAddrPtr());
1050  pktinfo.setRemoteSocketAddress(remoteAddr);
1051  }
1052  if (socket->getLocalSocketAddress().getFamily() == AF_INET) {

1077  }
1078

1082  _mcsocketMutex.unlock();
